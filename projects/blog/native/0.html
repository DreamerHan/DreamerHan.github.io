# get 请求处理

TIP

使用 queryString 做参数处理

const http = require("http");
// nodejs 自带的模块
// 将 key1=value1&key2=value2 形式转换成 {key1: value, key2: value2 ...}
const queryString = require("querystring");

const server = http.createServer((req, res) => {
  const { method, url } = req;

  // method: GET  url: 请求的完整路径
  console.log("method", method, "url", url);

  // url: 请求的完整路径 localhost:3000/api/blog/detail
  // url 即为：/api/blog/detail
  // 注意，每次都会挂上 /favicon.ico 的请求
  console.log("url", url);

  // query: req.query 本身是不存在的，为 undefined
  console.log("query", req.query);

  // req.query => { key1: value1, key2: value2 }
  req.query = queryString.parse(url.split("?")[1]);

  // res.end 需要返回字符串
  // 之所以使用 end 是因为 res 是一个流(stream)
  res.end(JSON.stringify(req.query));
});

server.listen(3000, () => {
  console.log("server is running at port 3000");
});

# post 请求处理

TIP

req.on('data', (data) => { 来接受 post 传输的数据 })

const http = require("http");

const server = http.createServer((req, res) => {
  if (req.method === "POST") {
    // content-type 数据类型： application/json
    console.log("content-type", req.headers["content-type"]);

    let postData = "";

    req.on("data", (chunk) => {
      console.log(typeof chunk); // object
      postData += chunk.toString();
    });

    req.on("end", () => {
      // postData { key1: value1, key2: value2 }
      // typeof  string
      console.log("postData", postData, typeof postData);

      res.end("Hello, World!!" + postData);
    });
  }
});

server.listen(3000, () => {
  console.log("server is running at port 3000");
});

# 整合 get 和 post

const http = require("http");
const queryString = require("querystring");

const server = http.createServer((req, res) => {
  const { url, method } = req;

  const [path, queryPart] = url.split("?");
  const query = queryString.parse(queryPart);

  // 这里结合下面的 res.end 说明
  // res.end 包裹的内容必须是字符串
  // setHeader 这里指定的内容格式，是说明 res.end 的字符串形式下中的内容是一个 json 的形式
  res.setHeader("Content-type", "application/json");

  // 返回的数据内容
  const resData = {
    url,
    method,
    path,
    query,
  };

  if (method === "GET") {
    res.end(JSON.stringify(resData));
  }

  if (method === "POST") {
    let postData = "";

    req.on("data", (data) => {
      postData += data;
    });

    req.on("end", () => {
      resData.postData = postData;

      res.end(JSON.stringify(resData));
    });
  }
});

server.listen(3000, () => {
  console.log("server is running at port 3000");
});
