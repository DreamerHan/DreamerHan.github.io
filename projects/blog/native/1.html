# 基础环境的搭建

# 服务的启动脚本

bin/www.js 单独抽离出了 http 启动服务的代码逻辑，将 req/res 的各种查询信息及逻辑处理进行模块化抽离。

const http = require("http");

// 这里将对 req/res 的具体处理进行模块化业务逻辑抽离
const serverHandler = require("../server/app");

const server = http.createServer(serverHandler);

server.listen(3000, () => {
  console.log("Server is running at port 3000");
});

# app.js

app.js 作为整个应用程序的业务逻辑入口，会进行以下逻辑的嵌套处理：

  1. get 请求参数的处理

    • req.path 存储请求的路径信息
    • req.query 存储请求的 query 信息

  2. post 请求中的参数处理，存储在 req.body 中

  3. cookie 的解析，存储在 req.cookie 对象中

  4. 路由逻辑的引用处理

TIP

req.path、req.query、req.body、req.cookie 初始时均为 undefined

const getQueryData = require("./utils/getQueryData");
const getPostData = require("./utils/getPostData");

module.exports = async (req, res) => {
  // 设置数据返回格式为 json 形式
  res.setHeader("Content-type", "application/json; text/plain;charset=utf-8");

  // req 上添加了 path 和 query 参数
  getQueryData(req);

  await getPostData(req);

  // 这里是各种路由的引用处理
  // ...

  // 未找到数据时 404
  res.setHeader(404);
  res.write("404 Not Found\n\r 这是404页面呀，老铁~~");
  res.end();
};

# utils/getQueryData.js

TIP

querystring 模块是 nodejs 中的原生模块

parse 方法可以将 key1=val1&key2=val2 的形式解析成 json 形式 { key1: val1, key2: val2 }

const querystring = require("querystring");

module.exports = (req) => {
  const [path, query] = req.url.split("?");

  req.path = path;

  // [注意] 如果没有query参数，req.query 会为 undefined
  // 后续从 req.query 结构赋值参数时，undefined 会导致报错的，所以最好设置默认值为 {}
  req.query = query ? querystring.parse(query) : {};
};

# utils/getPostData.js

module.exports = (req) => {
  return new Promise((resolve, reject) => {
    let postData = "";
    const { method, headers } = req;

    if (method === "POST" && headers["content-type"] === "application/json") {
      req.on("data", (data) => {
        postData += data.toString();
      });

      req.on("end", () => {
        req.body = postData ? JSON.parse(postData) : {};

        resolve();
      });
    } else {
      req.body = {};
      resolve();
    }
  });
};
