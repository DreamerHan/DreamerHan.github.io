说一下个人理解：

  1. 鉴于部分中间件，这个部分就当是 koa-json吧 🌚🌚🌚。就鉴于它的写法，包上一层 Promise.resolve() 着实可以避免报错，毕竟干程序猿 🐵 的，大多都是杠头，当然我不是 🌝🌝🌝，万一人家的中间件就不是 async...await的异步，比如像 koa-json(对不住了，铁子...🙈🙈🙈)，那碰到了koa-json报错是必然的，估计这也是讲师说报错的原因吧~

  2. 是鉴于有同样问题的同学得到的回复的不认可(回复是：在不用 async 不用 promise 的中间件中加一段异步的逻辑，就不能按照预期的顺序执行了)，也是鉴于自己当时对 express的疑问的原因，琢磨了一下。

  • 首先，用了异步逻辑，那必然是要用async...await的，不然逻辑肯定不对呀。所以看完回复，我瞬间又懵了。接着有了自己的答案
  • 包 Promise.resolve() 是为了避免报错，纯粹就是为了避免报错，和别的无关，看下面的代码就可以了
  • 所谓的执行顺序 其实重点是 next() 函数的前后执行顺序，只要你所有的中间件都是 async...await的，就算不用 Promise.resolve()包括也无所谓，OK 的。但是就是你用了 Promise.resolve() 包括，假设两个中间件 A 和 B，B中包含了异步逻辑，A 不使用 async...await的话，执行顺序依然不 OK。

# likeKoa 中间件












 
 
 
 
 


















































const http = require("http");

// 组合中间件
function compose(middlewares) {
  // ctx 就是外围定义的 ctx
  return function (ctx) {
    function dispatch(i) {
      const fn = middlewares[i];

      // dispath 的重复调用其实就是 next 函数的执行
      try {
        // 1.包 promise 的写法
        return Promise.resolve(fn(ctx, dispatch.bind(null, i + 1)));

        // 2.不包的写法
        // return fn(ctx, dispatch.bind(null, i + 1));
      } catch (err) {
        return Promise.reject(err);
      }
    }

    return dispatch(0);
  };
}

class LikeKoa2 {
  constructor() {
    this.middlewares = [];
  }

  use(fn) {
    this.middlewares.push(fn);
    // 返回当前实例，可以使用 app.use().use()的莲师调用
    return this;
  }

  createContext(req, res) {
    const ctx = {
      req,
      res,
    };

    return ctx;
  }

  handleRequest(ctx, fn) {
    return fn(ctx);
  }

  callback() {
    const fn = compose(this.middlewares);
    return (req, res) => {
      const ctx = this.createContext(req, res);

      return this.handleRequest(ctx, fn);
    };
  }

  listen(...args) {
    const server = http.createServer(this.callback());
    server.listen(...args);
  }
}

module.exports = LikeKoa2;

# 含 Promise.resolve 的测试

测试的中间件(cookie 设置)的下一个，是一个异步的中间件。当前就算是源码里有了 Promise.resolve()，顺序依然是乱的。因为你 cookie 里没有 async...await，这才是关键。

执行得到的结果是： koa-json111 -> 1 -> 2 -> 3 -> 6 -> 7 -> koa-json222 -> 4 -> 5


































 
 
 
 
 
 
 
 
 
 
 


















const Koa = require("./likeKoa2");
const json = require("koa-json");

const app = new Koa();

function getPostData(ctx) {
  return new Promise((resolve) => {
    const timer = setTimeout(() => {
      ctx.body = {
        username: "test",
        password: "123456",
      };

      clearTimeout(timer);
      resolve();
    }, 3000);
  });
}

// 日志
app.use(json());

app.use(async (ctx, next) => {
  console.log("1 like koa2 服务请求开始...");

  await next();

  console.log("7 like koa2 服务请求结束...");
});

// 日志 你不包 promise.resolve() 放这，它绝对报错
// app.use(json());

app.use((ctx, next) => {
  console.log("2 设置 cookie");
  ctx.cookie = {
    userid: "123456",
  };

  next();

  console.log("6 cookie 设置结束");
});

app.use(async (ctx, next) => {
  console.log("3 postdata 请求处理...");

  console.log("5 postdata 请求结束...");

app.use((ctx) => {
  console.log("4 like koa2 执行了到这里...");

app.listen(7788, () => {

# 不含 Promise.resolve 的测试

如果，此时 likeKoa2 源码里的不使用 Promise.resolve() 的写法，直接 return 函数执行， 如下：

// dispath 的重复调用其实就是 next 函数的执行
try {
  // 1.包 promise 的写法
  // return Promise.resolve(fn(ctx, dispatch.bind(null, i + 1)));

  // 2.不包的写法
  return fn(ctx, dispatch.bind(null, i + 1));
} catch (err) {
  return Promise.reject(err);
}

中间件里依然不使用 async...await，执行的顺序仍然是： koa-json111 -> 1 -> 2 -> 3 -> 6 -> 7 -> koa-json222 -> 4 -> 5

于是，如果加上 async...await，

// 注意哈，这里我放开了注释，上面还有同样的一行就要注释掉
// 日志 你不包 promise.resolve() 放这，它绝对报错

// 这个就用来测试
app.use(async (ctx, next) => {
  console.log("2 设置 cookie");
  ctx.cookie = {
    userid: "123456",
  };

  await next();

  console.log("6 cookie 设置结束");
});

执行顺序为： koa-json111 -> 1 -> 2 -> 3 -> 4 -> 5 -> 6 -> 7 -> koa-json222

而且，不报错~😎😎😎
