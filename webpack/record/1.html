# 函数 __webpack_require__

这要从一个简单的 webpack 构建流程说起：

  1. 通过 yarn init -y 来初始化一个项目
  2. yarn add webpack webpack-cli 安装 webpack 的基础依赖
  3. 在 src 文件夹并在下面创建 index.js 和 test.js 文件
// test.js -- 暴露函数
function test() {
  console.log("test 函数");
}
module.exports = test;

// index.js -- 引入函数并执行
const testFn = require("./test");
testFn();

# 打包后的代码

我们要关注的就是构建后的文件，下列文件已经删除来不必要的注释，并对必要的注释进行了翻译 😊

(() => {
  // webpackBootstrap
  var __webpack_modules__ = {
    "./src/index.js": (
      __unused_webpack_module,
      __unused_webpack_exports,
      __webpack_require__
    ) => {
      eval(
        'const testFn = __webpack_require__(/*! ./test */ "./src/test.js")\n\ntestFn()\n\n\n\n//# sourceURL=webpack://webpack-1/./src/index.js?'
      );
    },

    "./src/test.js": (module) => {
      eval(
        "function test() {\n  console.log('test 函数')\n}\n\nmodule.exports = test\n\n//# sourceURL=webpack://webpack-1/./src/test.js?"
      );
    },
  };
  // The module cache -- 模块缓存
  var __webpack_module_cache__ = {};

  // The require function -- 请求函数
  function __webpack_require__(moduleId) {
    // Check if module is in cache
    var cachedModule = __webpack_module_cache__[moduleId];
    if (cachedModule !== undefined) {
      return cachedModule.exports;
    }
    // Create a new module (and put it into the cache) -- 创建一个新的模块，并把它放模块缓存中
    var module = (__webpack_module_cache__[moduleId] = {
      // no module.id needed -- 不再需要 module.id
      // no module.loaded needed -- 不再需要 module.loaded
      exports: {},
    });

    // Execute the module function -- 执行模块的函数
    __webpack_modules__[moduleId](module, module.exports, __webpack_require__);

    // Return the exports of the module -- 返回模块的导出数据
    return module.exports;
  }

  // startup -- 启动函数
  // Load entry module and return exports -- 加载入口模块并返回导出
  // This entry module can't be inlined because the eval devtool is used.
  // 入口模块无法进行内联处理，因为使用了 eval devtool
  var __webpack_exports__ = __webpack_require__("./src/index.js");
})();

可以看到，webpack 构建后的模块加载系统其实很简单，它是一个自执行函数。函数体结构基本有以下几个模块：

  1. 构建模块形成的 map ，赋值给了__webpack_modules__，如果将这个 map 进行结构简化后是这样的
var __webpack_modules__ = {
  模块path1: 模块执行函数1,
  模块path2: 模块执行函数2,
};
  1. 定义一个用来缓存模块的对象 __webpack_module_cache__
  2. 定义一个模块加载函数 __webpack_require__()
  3. 使用 __webpack_require__() 加载入口模块文件

单从代码的多少来看就知道 __webpack_require__() 绝对就是大 Boss。它接收一个 moduleId 参数，这个 moduleId 其实叫 modulePath 更合适，它就是文件的路径。

# __webpack_require__ 执行步骤

__webpack_require__() 的执行步骤如下：

  1. 判断当前要获取的这个模块是否已经在缓存中，如果已经缓存就返回该模块的 exports 对象，就是 module.exports
// Check if module is in cache
var cachedModule = __webpack_module_cache__[moduleId];
if (cachedModule !== undefined) {
  return cachedModule.exports;
}
  1. 如果没有缓存，就新建一个模块 module，同时放入缓存中。注意 webpack5 中，已经不再需要 id 和 loaded 属性了。
// Create a new module (and put it into the cache) -- 创建一个新的模块，并把它放模块缓存中
var module = (__webpack_module_cache__[moduleId] = {
  // no module.id needed -- 不再需要 module.id
  // no module.loaded needed -- 不再需要 module.loaded
  exports: {},
});
  1. 从开始的模块映射 map 中取出该模块路径对应的函数并执行。
// Execute the module function -- 执行模块的函数
__webpack_modules__[moduleId](module, module.exports, __webpack_require__);

// 可以这么理解，这个函数执行后，这一步就有了一个结果是 module.exports = xxx
  1. 返回模块的 exports 数据

# 解析模块执行函数

__webpack_require__() 接收三个参数，module、module.exports 和 __webpack_require__函数

其中 module、module.exports 的作用和 CommonJS 中的 module、module.exports 的作用是一样的，而 __webpack_require__ 相当于 CommonJS 中的 require。

自执行函数的最后，以入口函数 ./src/index.js 为参数开始执行。

// startup -- 启动函数
// Load entry module and return exports -- 加载入口模块并返回导出
// This entry module can't be inlined because the eval devtool is used.
// 入口模块无法进行内联处理，因为使用了 eval devtool
var __webpack_exports__ = __webpack_require__("./src/index.js");

来分析一下每个模块对应的 eval 函数内容

# 入口文件 index.js

// 美化后的 index.js
const testFn = __webpack_require__(/*! ./test */ \"./src/test.js\")

testFn()

//# sourceURL=webpack://webpack-1/./src/index.js?

------------------------------------------------------------------------------------

// 原始的 index.js
const testFn = require('./test')
testFn()

对比构建之前的 index.js 文件，除了最后一行的注释，就是把 require 函数变成了 __webpack_require__ 函数

# 暴露模块的 test.js

// 美化后的 test.js
function test() {
  console.log('test 函数')
}

module.exports = test

//# sourceURL=webpack://webpack-1/./src/test.js

------------------------------------------------------------------------------------

// 原始的 test.js
function test() {
  console.log('test 函数')
}
module.exports = test

test.js 文件几乎没有变化，当然现在还有添加上各种 babel、es6 转 es5 的规范等等。

# 还原模块执行顺序

从 __webpack_require__ 函数执行的步骤 3 中可知，__webpack_require__ 函数加载一个模块时，会直接调用该模块对应的函数，然后返回该模块的 exports 对象。

根据该思路来推断 index.js 和 test.js 的执行顺序如下：

自执行函数执行入口文件开始执行：

var __webpack_exports__ = __webpack_require__("./src/index.js");

__webpack_require__ 开始调用 ./src/index.js 对应的函数：

"./src/index.js":
    ((__unused_webpack_module, __unused_webpack_exports, __webpack_require__) => {

      // eval("const testFn = __webpack_require__(/*! ./test */ \"./src/test.js\")\n\ntestFn()\n\n\n\n//# sourceURL=webpack://webpack-1/./src/index.js?");

      const testFn = __webpack_require__(/*! ./test */ \"./src/test.js\")

      testFn()

      //# sourceURL=webpack://webpack-1/./src/index.js?
    }),

当执行到加载 test.js 模块时

const testFn = __webpack_require__(/*! ./test */ \"./src/test.js\")

又开始加载 ./src/test.js 对应的函数，该函数只接收一个 module 参数：

"./src/test.js":
    ((module) => {

      // 参数module: { exports: {} }

      // eval("function test() {\n  console.log('test 函数')\n}\n\nmodule.exports = test\n\n//# sourceURL=webpack://webpack-1/./src/test.js?");

      function test() {
        console.log('test 函数')
      }

      module.exports = test

      //# sourceURL=webpack://webpack-1/./src/test.js
    })

而 module.exports 正好等于 test 函数。所以，再倒推回去，index.js 的 __webpack_require__ 函数加载到的 test.js 返回的 exports 正好就是 test 函数。于是，test 函数就可以在 index.js 中执行了。
